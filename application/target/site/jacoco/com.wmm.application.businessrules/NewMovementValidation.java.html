package com.wmm.application.businessrules;

import com.wmm.domain.IncompleteRequestDataException;
import com.wmm.domain.entity.Movement;
import lombok.RequiredArgsConstructor;

import static com.wmm.application.exceptions.ExceptionMessages.*;

@RequiredArgsConstructor
public class NewMovementValidation {

    private final Movement movement;

    public void check() {
        if (movement.getDate() == null) {
            throw new IncompleteRequestDataException(REQUEST_DATE_WRONG);
        }
        if (movement.getAmount() < 1) {
            throw new IncompleteRequestDataException(REQUEST_AMOUNT_ZERO);
        }

        if (movement.getUserId()==null || movement.getUserId().isBlank() || movement.getUserId().isEmpty()) {
            throw new IncompleteRequestDataException(REQUEST_AMOUNT_ZERO);
        }

        if (movement.getDetail()==null || movement.getDetail().isBlank() || movement.getDetail().isEmpty()) {
            throw new IncompleteRequestDataException(REQUEST_DETAIL_EMPTY);
        }

        if (movement.getMovementType()==null ||movement.getMovementType().equals("") ) {
            throw new IncompleteRequestDataException(REQUEST_MOVEMENT_TYPE_EMPTY);
        }

        if (movement.getCreatedAt()==null) {
            throw new IncompleteRequestDataException(CREATED_AT_NOT_SETTLED);
        }


    }
}
